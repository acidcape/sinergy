import random

def play_guessing_game():
    # Генерируем случайное число от 1 до 100, которое нужно угадать
    secret_number = random.randint(1, 100)
    max_attempts = 10  # Максимальное количество попыток

    # Приветственное сообщение
    print("Приветствуем в игре 'Угадай число'!")
    print("Я загадал число в диапазоне от 1 до 100.")
    print(f"Попробуйте угадать его за {max_attempts} попыток.")

    attempt = 1  # Счётчик попыток

    # Цикл продолжается, пока не исчерпаны попытки
    while attempt <= max_attempts:
        # Получаем ввод от пользователя
        user_input = input(f"Попытка {attempt}: Ваш вариант: ")
        
        try:
            # Преобразуем ввод в целое число
            guess = int(user_input)
            
            # Проверяем, находится ли число в допустимом диапазоне
            if not 1 <= guess <= 100:
                print("Число должно быть в пределах от 1 до 100.")
                continue  # Пропускаем оставшийся код итерации и переходим к следующей попытке

            # Сравниваем ввод пользователя с загаданным числом
            if guess == secret_number:
                print(f"Отлично! Вы угадали число {secret_number} с {attempt}-й попытки.")
                break  # Завершаем цикл, если пользователь угадал
            elif guess < secret_number:
                print("Маловато будет.")  # Подсказка: число больше
            else:
                print("Перебор.")  # Подсказка: число меньше

            attempt += 1  # Увеличиваем счётчик попыток

        except ValueError:
            # Обработка случая, если пользователь ввёл не число
            print("Введите, пожалуйста, целое число.")

    else:
        # Этот блок выполнится, если пользователь не угадал число за все попытки
        print(f"Попытки закончились. Загаданное число было: {secret_number}")

# Точка входа в программу
if __name__ == "__main__":
    play_guessing_game()
