import math  # Импортируем модуль math для использования функции factorial

# Функция для получения неотрицательного целого числа от пользователя
def get_non_negative_integer():
    value = input("Введите неотрицательное целое число: ")  # Запрашиваем ввод у пользователя
    if not value.isdigit():
        # Проверка: если строка не состоит только из цифр, вызываем исключение
        raise ValueError("Введено нецелое или отрицательное число.")
    return int(value)  # Преобразуем строку в целое число и возвращаем

# Основная функция программы
def main():
    try:
        number = get_non_negative_integer()  # Получаем корректное число от пользователя
        factorial = math.factorial(number)  # Вычисляем факториал с помощью встроенной функции
        print(f"Факториал числа {number} равен {factorial}")  # Выводим результат
    except ValueError as error:
        # Обрабатываем исключения, связанные с некорректным вводом
        print(f"Ошибка: {error}")

# Точка входа в программу
if __name__ == "__main__":
    main()
